    创建对象的三种方式

    # 创建对象的三种方式

    	1.字面量
  var obj = {name: '三毛'}
	2.new Object
	var obj = new Object()
	obj.name = name
  以上两种一次只能创建一个对象
	3.构造函数：
  将对象中相同得属性和方法抽象出来，封装到函数中	
    //1.申明构造函数
    function 构造函数名(){
      this.属性 = 值
      this.方法 = function (){}
    }
    //2.调用构造函数
    new 构造函数名()

    # 调用对象

    1.对象名.属性名
2.对象名['属性名']

    # new关键字执行过程

    1.创建一个空的对象
2.this指向
3.执行代码，添加属性和方法
4.返回这个对象

    # 对象分类

    	1.自定义对象
	2.内置对象：JS语言自带的一些对象
		以上两个属于ECMAScript
	3.浏览器对象（JS独有）

  3.1Math对象
	Math.max() //返回-Infinity 什么不写
	Math.PI
	Math.abs() //绝对值
	Math.floor()//向下取整
	Math.round()//四舍五入 
  注意	Math.round(-1.5) //-1
	Math.random() // 0 <= x < 1

  //随机获取两个数之间的整数（含两端）
    function getRandomInt(min, max) {
      min = Math.ceil(min);
      max = Math.floor(max);
      return Math.floor(Math.random() * (max - min + 1)) + min; //含最大值，含最小值
    }
   //随机点名
    var arr = ['毛','毛毛','毛毛毛']
    console.log(arr[getRandomInt(0,arr.length - 1)])

  3.2 日期对象:
	var date = new Date() //返回当前时间 什么不写
	参数的常见写法 2021,5,19  //容易错，5月变6月
	或者是字符串 '2021-5-19 15:31:31' //常用
  日期对象的方法

  日期的总的毫秒形式：距离1970.1.1过了多少毫秒
	Date对象是基于1970年1月1日（世界标准时间）起的毫秒数
	1.var date = newDate()
		date.valueOf() 或者 date.getTime()
	2.简单写法 最常用 
		var date1 = +new Date()
	3.H5新增 
		console.log(Date.now())

    # 对象创建模型

    方法1	Object构造函数模式
	*先创建空obj,再添加属性
	*适用于不确定对象内部数据
	*问题：语句太多
  ex.
  var p = new Object()
  p.name = 'Tome'
  p.age = 'age'

方法2	对象字面量模式
	*一条语句
	*使用{} 同时指定属性
	*适用于内部数据确定
	*问题：如果需要创建多个，有重复代码

  var p = {
    name: 'Tome',
    age:= 'age'
  }

方法3	工厂模式
	*通过工厂函数（返回一个对象的函数）创建对象并返回
	*适用于创建多个对象
	*问题：对象没有一个具体的类型，都是Obj

  function createP(name,age) {
    var obj = {
      name: name,
      age: age,
      setName: function (){
        this.name = name
      }
    }
    return obj
  }
  var p1 = createP('Tome',12)
  var p2 = createP('Bob',13)
  问题p1和p2都是obj

  function createStudent(name,age,price) {
    var obj = {
      name: name,
      age: age,
      price:price
      setName: function (){
        this.name = name
      }
    }
    return obj
  }
  var s1 = createStudent('Tim',12,12000)
  问题s1也是obj

  以上三个对象没有具体类型

  方法4	自定义构造函数模式
	*通过new创建对象
	*场景：需要创建多个类型确定的对象
	*问题：每个对象都有相同的数据（方法），浪费内存
	*将这个方法放到原型中！方法5 
  //定义构造函数
  function Person(name,age) {
    this.name = name
    this.age = age
    this.setName = function (name){
      this.name = name
    }
  }
  //创建对象
	var p1 = new Person('Bob',99)
	console.log(p1 instanceof Person)

  //定义构造函数 学生
  function Student(name,age) {
    this.name = name
    this.age = age
    this.setName = function (name){
      this.name = name
    }
  }
	var s1 = new Student('Tim',22)

  方法5	构造函数+原型
  function Person(name,age) { //只初始化一般属性
    this.name = name
    this.age = age
  }

  Person.prototype.setName = function(name){
      this.name = name //this是实例
  }

	var p1 = new Person('Bob',99)
